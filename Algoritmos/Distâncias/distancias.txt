distancias_vertice(G = (V, E), v)
    para todo v* ∈ V
        cor[v*] = branco
        d[v*] = infinito
    fim para

    d[v] = 0
    cor[v] = cinza
    f.insere(v)

    enquanto !f.vazio()
        u = f.remove()

        para todo u* ∈ N(u)
            se cor[u*] = branco
                d[u*] = d[u] + 1
                cor[u*] = cinza
                f.insere(u*)
            fim se
        fim para

        cor[u] = preto
    fim enquanto

    retornar d[ ]
fim algoritmo
