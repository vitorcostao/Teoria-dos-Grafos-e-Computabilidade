DFS_Recursivo(grafo)

	para todo v e V
		cor[v] = branco
	fim para

	para todo v e V
		se cor[v] = branco
			DFS_Visita(grafo, v)
		fim se
	fim para
fim algoritmo

DFS_Visita(grafo, v)
	cor[v] = cinza
	para todo u e N(v)
		se cor[u] = branco
			DFS_Visita(grafo, u)
		senão se cor[v*] = cinza e v* != pai
			// Detectou um ciclo
		fim se
	fim para 
	cor[v] = preto
fim algoritmo
