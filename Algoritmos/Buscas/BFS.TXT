BFS(grafo, alvo) -> Retorna booleano

	para todo v e V
		cor[v] = branco
	fim para

	resp = false

	para todo v e V
		se cor[v] = branco
			f.insere(v)
			cor[v] = cinza
			enquanto !f.vazio()
				u = f.remove()
				se u == alvo
					resp = true
					retorna resp
				fim se
				para todo u* e N(u)
					se cor[u*] = branco
						f.insere(u*)
						cor[u*] = cinza
					fim se
				fim para
				cor[u] = preto
			fim enquanto
		fim se
	fim para
	retorna resp
fim algoritmo
